# steps to push files to the git

 git init  // Initialized empty Git repository
 ls -a // list of files 
 git status
 git add filename 
 git log 
 git diff text2.txt //check the difference
 git checkout text2.txt^C //undo
 git remote add <name> https://github.com/shodanshetty/test.git 
 git push -u <remote name> <branch name>
